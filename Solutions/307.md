# Description:

Given an integer array nums, find the sum of the elements between indices i and j (i â‰¤ j), inclusive.

The update(i, val) function modifies nums by updating the element at index i to val.

**Example:**

```
Given nums = [1, 3, 5]

sumRange(0, 2) -> 9
update(1, 2)
sumRange(0, 2) -> 8
```

**Note:**

1. The array is only modifiable by the update function.
2. You may assume the number of calls to update and sumRange function is distributed evenly.

# Idea:
We can use segment tree to solve the problem. (I will demonstrate this later... if I cam remember this. haha)

# Solution:

- Python:

```python
class NumArray(object):

    def __init__(self, nums):
        """
        :type nums: List[int]
        """
        self.len = len(nums)
        self.cnt = 0
        self.sum = [0] * (5 * self.len)
        self.build(1, self.len, 1, nums)
        
    def pushUp(self, rt):
        self.sum[rt] = self.sum[rt << 1] + self.sum[rt << 1 | 1]
    
    def build(self, l, r, rt, nums):
        if l > r:
            return
        if l == r:
            self.sum[rt] = nums[self.cnt]
            self.cnt += 1
            return
        else:
            m = (l + r) >> 1
            self.build(l, m, rt << 1, nums)
            self.build(m + 1, r, rt << 1 | 1, nums)
            self.pushUp(rt)
            
    def selfupdate(self, p, add, l, r, rt):
        if l > r:
            return
        if l == r:
            self.sum[rt] = add
            return
        m = (l + r) >> 1
        if p <= m:
            self.selfupdate(p, add, l, m, rt << 1)
        else:
            self.selfupdate(p, add, m + 1, r, rt << 1 | 1)
        self.pushUp(rt)
        
    
    def query(self, L, R, l, r, rt):
        if L <= l and r <= R:
            return self.sum[rt]
        ret = 0
        m = (l + r) >> 1
        if L <= m:
            ret += self.query(L, R, l, m, rt << 1)
        if R > m:
            ret += self.query(L, R, m + 1, r, rt << 1 | 1)
        return ret
        
    def update(self, i, val):
        """
        :type i: int
        :type val: int
        :rtype: void
        """
        self.selfupdate(i + 1, val, 1, self.len, 1)
        

    def sumRange(self, i, j):
        """
        :type i: int
        :type j: int
        :rtype: int
        """
        return self.query(i + 1, j + 1, 1, self.len, 1)
        


# Your NumArray object will be instantiated and called as such:
# obj = NumArray(nums)
# obj.update(i,val)
# param_2 = obj.sumRange(i,j)
```

- C++:

```cpp
class NumArray {

#define lson l, m, rt << 1
#define rson m + 1, r, rt << 1 | 1
    
public:
    int cnt;
    int len;
    vector<int> sum;
    
    void pushUp(int rt) {
        sum[rt] = sum[rt << 1] + sum[rt << 1 | 1];
    }
    
    void build(int l, int r, int rt, vector<int> & nums) {
        if (l > r) {
            return ;
        }
        if (l == r) {
            sum[rt] = nums[cnt++];
            return ;
        }
        int m = (l + r) >> 1;
        build(lson, nums);
        build(rson, nums);
        pushUp(rt);
    }
    
    void update(int p, int add, int l, int r, int rt) {
        if (l > r) {
            return ;
        }
        if (l == r) {
            sum[rt] = add;
            return ;
        }
        int m = (l + r) >> 1;
        if (p <= m) update(p, add, lson);
        else update(p, add, rson);
        pushUp(rt);
    }
    
    int query(int L, int R, int l, int r, int rt) {
        if (L <= l and r <= R) {
            return sum[rt];
        }
        int m = (l + r) >> 1;
        int ret = 0;
        if (L <= m) ret += query(L, R, lson);
        if (R > m) ret += query(L, R, rson);
        return ret;
    }
    
    NumArray(vector<int> nums) {
        len = nums.size();
        // initialize
        for (int i = 0; i < len * 5; ++i) {
            sum.push_back(0);
        }
        cnt = 0;
        build(1, len, 1, nums);
    }
    
    void update(int i, int val) {
        update(i + 1, val, 1, len, 1);
    }
    
    int sumRange(int i, int j) {
        return query(i + 1, j + 1, 1, len, 1);
    }
};

/**
 * Your NumArray object will be instantiated and called as such:
 * NumArray obj = new NumArray(nums);
 * obj.update(i,val);
 * int param_2 = obj.sumRange(i,j);
 */
```
