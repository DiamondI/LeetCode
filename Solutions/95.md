# [95. Unique Binary Search Trees II](https://leetcode.com/problems/unique-binary-search-trees-ii/description/)

## Description

Given an integer *n*, generate all structurally unique **BST's** (binary search trees) that store values *1* ... *n*.

**Example:**

```
Input: 3
Output:
[
  [1,null,3,2],
  [3,2,null,1],
  [3,1,null,null,2],
  [2,1,3],
  [1,null,2,null,3]
]
Explanation:
The above output corresponds to the 5 unique BST's shown below:

   1         3     3      2      1
    \       /     /      / \      \
     3     2     1      1   3      2
    /     /       \                 \
   2     1         2                 3
```

## Idea

We can solve this question iteratively as shown below.

## Solution

### C++

```cpp
/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    vector<TreeNode*> generateTrees(int n) {
        vector<TreeNode*> res;
        if (n <= 0) {
            return {};
        }
        return buildTrees(1, n);
    }
    
    TreeNode* createNode(int val, TreeNode* &left, TreeNode* &right) {
        TreeNode *node = new TreeNode(val);
        node->left = left;
        node->right = right;
        return node;
    }
    
    vector<TreeNode*> buildTrees(int first, int last) {
        vector<TreeNode*> res;
        if (first > last) {
            return {NULL};
        }
        for (int root = first; root <= last; ++root) {
            for (auto left : buildTrees(first, root - 1)) {
                for (auto right : buildTrees(root + 1, last)) {
                    res.push_back(createNode(root, left, right));
                }
            }
        }
        return res;
    }
};
```

### Python 3

```python
# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def generateTrees(self, n):
        """
        :type n: int
        :rtype: List[TreeNode]
        """
        if n == 0:
            return []
        def createNode(val, left, right):
            node = TreeNode(val)
            node.left = left
            node.right = right
            return node
        def buildTree(first, last):
            if first > last:
                return [None]
            return [createNode(root, left, right)
                    for root in range(first, last + 1)
                    for left in buildTree(first, root - 1)
                    for right in buildTree(root + 1, last)]
        return buildTree(1, n)
```
